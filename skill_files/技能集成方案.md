### 步骤3：技能调用接口

```python
# /root/.openclaw/skills/skill_manager.py
import importlib
import os

class SkillManager:
    def __init__(self):
        self.skills = {}
        self._load_skills()

    def _load_skills(self):
        """加载所有技能"""
        skill_dir = "/root/.openclaw/skills/"
        for skill_name in os.listdir(skill_dir):
            skill_path = os.path.join(skill_dir, skill_name)
            if os.path.isdir(skill_path):
                try:
                    # 加载技能模块
                    module = importlib.import_module(f"skills.{skill_name}")
                    self.skills[skill_name] = module
                except Exception as e:
                    print(f"加载技能失败: {skill_name}, {e}")

    def execute_skill(self, skill_name, task):
        """执行技能"""
        if skill_name in self.skills:
            skill = self.skills[skill_name]
            return skill.execute(task)
        else:
            raise ValueError(f"技能不存在: {skill_name}")

    def get_skill_by_category(self, category):
        """根据分类获取技能"""
        return {name: skill for name, skill in self.skills.items()
                if skill.get('category') == category}
```

### 决策逻辑

```python
# /root/.openclaw/skills/decision_engine.py
from typing import List, Dict
import re

class DecisionEngine:
    """智能决策引擎"""

    def __init__(self, skill_manager):
        self.skill_manager = skill_manager
        self.rules = {
            "量化分析": ["12-量化交易V2.2", "07-互联网访问", "03-图像识别"],
            "内容创作": ["02-联网搜索", "09-AI编程", "04-自动迭代",
                      "05-内容总结", "06-AI总结"],
            "数据采集": ["07-互联网访问", "01-长期记忆"],
            "内容发布": ["小红书发布"],
            "学术研究": ["02-联网搜索", "10-学术写作", "04-自动迭代"],
        }

    def analyze_task(self, task: str) -> List[str]:
        """分析任务并推荐技能"""
        # 关键词匹配
        for pattern, skills in self.rules.items():
            if re.search(pattern, task):
                return skills

        # 默认：AI编程 + 自动迭代
        return ["09-AI编程", "04-自动迭代"]

    def get_confidence(self, task: str) -> float:
        """获取决策置信度"""
        # 简单实现：关键词匹配度
        max_matches = 0
        for pattern, skills in self.rules.items():
            matches = len(re.findall(pattern, task))
            max_matches = max(max_matches, matches)

        # 归一化到 0-1
        return min(max_matches / 3.0, 1.0)

    def should_auto_invoke(self, task: str, threshold: float = 0.7) -> bool:
        """是否自动调用技能"""
        confidence = self.get_confidence(task)
        return confidence >= threshold
```
