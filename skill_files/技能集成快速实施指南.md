```bash
# 创建管理器文件
cat > /root/.openclaw/skills/manager.py << 'EOF'
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
技能管理器
统一管理和调用所有技能
"""

import os
import importlib
from typing import List, Dict, Any

class SkillManager:
    """技能管理器"""

    def __init__(self):
        self.skills = {}
        self.skill_configs = {}
        self._load_skills()

    def _load_skills(self):
        """加载所有技能"""
        skill_dir = "/root/.openclaw/skills/"
        for skill_name in os.listdir(skill_dir):
            skill_path = os.path.join(skill_dir, skill_name)
            if os.path.isdir(skill_path):
                try:
                    # 读取配置
                    config_file = os.path.join(skill_path, "config.yaml")
                    if os.path.exists(config_file):
                        import yaml
                        with open(config_file, 'r') as f:
                            config = yaml.safe_load(f)
                            self.skill_configs[skill_name] = config

                    # 加载技能模块
                    module_path = os.path.join(skill_path, skill_name)
                    spec = importlib.util.spec_from_file_location(
                        skill_name,
                        os.path.join(module_path, "__init__.py")
                    )
                    module = importlib.util.module_from_spec(spec)
                    spec.loader.exec_module(module)

                    self.skills[skill_name] = {
                        'module': module,
                        'config': self.skill_configs.get(skill_name, {}),
                        'path': skill_path
                    }
                    print(f"✅ 加载技能成功: {skill_name}")
                except Exception as e:
                    print(f"❌ 加载技能失败: {skill_name}, {e}")

    def get_skill(self, skill_name: str):
        """获取技能"""
        return self.skills.get(skill_name)

    def execute_skill(self, skill_name: str, task: Dict[str, Any]):
        """执行技能"""
        skill = self.get_skill(skill_name)
        if skill:
            try:
                module = skill['module']
                if hasattr(module, 'execute'):
                    return module.execute(task)
                else:
                    print(f"⚠️ 技能没有execute方法: {skill_name}")
                    return None
            except Exception as e:
                print(f"❌ 执行技能失败: {skill_name}, {e}")
                return None
        else:
            print(f"⚠️ 技能不存在: {skill_name}")
            return None

    def list_skills(self) -> List[str]:
        """列出所有技能"""
        return list(self.skills.keys())

    def get_skills_by_category(self, category: str) -> List[str]:
        """根据分类获取技能"""
        skills = []
        for skill_name, skill_data in self.skills.items():
            config = skill_data.get('config', {})
            if config.get('category') == category:
                skills.append(skill_name)
        return skills

# 创建单例
skill_manager = SkillManager()

if __name__ == "__main__":
    # 测试
    print("已加载技能:")
    for skill_name in skill_manager.list_skills():
        print(f"  - {skill_name}")
EOF

chmod +x /root/.openclaw/skills/manager.py
```

```bash
# 创建决策引擎
cat > /root/.openclaw/skills/decision.py << 'EOF'
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
智能决策引擎
根据任务自动选择技能
"""

import re
from typing import List, Dict

class DecisionEngine:
    """智能决策引擎"""

    def __init__(self, skill_manager):
        self.skill_manager = skill_manager
        self.rules = {
            # 量化分析相关
            "量化": ["12-量化交易V2.2", "07-互联网访问", "03-图像识别"],
            "股票": ["12-量化交易V2.2", "07-互联网访问"],
            "K线": ["12-量化交易V2.2", "03-图像识别"],
            "回测": ["12-量化交易V2.2"],

            # 内容创作相关
            "文章": ["02-联网搜索", "09-AI编程", "04-自动迭代"],
            "发布": ["小红书发布"],
            "小红书": ["小红书发布"],

            # 数据分析相关
            "数据": ["07-互联网访问", "01-长期记忆"],
            "图表": ["03-图像识别", "09-AI编程"],

            # 研究相关
            "研究": ["02-联网搜索", "10-学术写作"],
            "论文": ["10-学术写作"],

            # 通用
            "搜索": ["02-联网搜索"],
            "图片": ["03-图像识别"],
            "总结": ["05-内容总结", "06-AI总结"],
        }

    def analyze_task(self, task: str) -> List[str]:
        """分析任务并推荐技能"""
        recommended_skills = []

        # 关键词匹配
        for pattern, skills in self.rules.items():
            if re.search(pattern, task, re.IGNORECASE):
                for skill in skills:
                    if skill not in recommended_skills:
                        recommended_skills.append(skill)

        # 如果没有匹配，使用默认技能
        if not recommended_skills:
            recommended_skills = ["09-AI编程", "04-自动迭代"]

        return recommended_skills

    def get_confidence(self, task: str) -> float:
        """获取决策置信度"""
        max_matches = 0
        for pattern, skills in self.rules.items():
            matches = len(re.findall(pattern, task, re.IGNORECASE))
            max_matches = max(max_matches, matches)

        # 归一化到 0-1
        return min(max_matches / 2.0, 1.0)

    def should_auto_invoke(self, task: str, threshold: float = 0.7) -> bool:
        """是否自动调用技能"""
        confidence = self.get_confidence(task)
        return confidence >= threshold

# 创建单例
from manager import skill_manager
decision_engine = DecisionEngine(skill_manager)

if __name__ == "__main__":
    # 测试
    test_tasks = [
        "分析股票600000的K线图",
        "写一篇关于量化交易的文章并发布到小红书",
        "研究最新的AI论文",
        "总结这段内容"
    ]

    for task in test_tasks:
        skills = decision_engine.analyze_task(task)
        confidence = decision_engine.get_confidence(task)
        print(f"\n任务: {task}")
        print(f"推荐技能: {skills}")
        print(f"置信度: {confidence:.2f}")
EOF

chmod +x /root/.openclaw/skills/decision.py
```

### 步骤4：创建主入口

```bash
# 创建主入口
cat > /root/.openclaw/skills/main.py << 'EOF'
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
技能系统主入口
统一管理和调用所有技能
"""

import sys
import json
from manager import skill_manager
from decision import decision_engine

def main():
    """主函数"""
    if len(sys.argv) < 2:
        print("使用方法: python main.py <task>")
        print("示例: python main.py '分析股票600000'")
        return

    # 获取任务
    task = sys.argv[1]

    # 智能决策
    recommended_skills = decision_engine.analyze_task(task)
    confidence = decision_engine.get_confidence(task)

    print("=" * 70)
    print(f"任务: {task}")
    print(f"推荐技能: {', '.join(recommended_skills)}")
    print(f"决策置信度: {confidence:.2f}")
    print("=" * 70)

    # 依次执行技能
    results = []
    for skill_name in recommended_skills:
        print(f"\n🎯 执行技能: {skill_name}")
        result = skill_manager.execute_skill(skill_name, {"task": task})
        results.append({
            "skill": skill_name,
            "result": result
        })
        print(f"✅ 执行完成")

    # 最终总结
    print("\n" + "=" * 70)
    print("执行总结")
    print("=" * 70)

    for item in results:
        print(f"\n技能: {item['skill']}")
        if item['result']:
            print(f"结果: {str(item['result'])[:100]}...")
        else:
            print("结果: None")

if __name__ == "__main__":
    main()
EOF

chmod +x /root/.openclaw/skills/main.py
```
